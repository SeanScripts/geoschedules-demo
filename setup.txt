CREATE TABLE employee (employee_id integer primary key autoincrement, first_name varchar(20), last_name varchar(30), email_address varchar(30), hourly_wage real, supervisor_id int, hash varchar(256), salt varchar(256));
CREATE TABLE supervisor (supervisor_id integer primary key autoincrement, first_name varchar(20), last_name varchar(30), email_address varchar(30), hourly_wage real, hash varchar(256), salt varchar(256));
CREATE TABLE schedule (schedule_id integer primary key autoincrement, employee_id int, site_id int, start_time datetime, end_time datetime, repeat varchar(10));
CREATE TABLE client (client_id integer primary key autoincrement, client_name varchar(30));
CREATE TABLE site (site_id integer primary key autoincrement, site_name varchar(50), client_id int, address varchar(50), city varchar(20), state CHAR(2), zip CHAR(5), hourly_service_fee real, latitude real, longitude real);
CREATE TABLE shift (shift_id integer primary key autoincrement, employee_id int, site_id int, clock_in_time datetime, clock_out_time datetime, latitude_in real, longitude_in real, latitude_out real, longitude_out real);
CREATE TABLE subsite (subsite_id integer primary key autoincrement, subsite_name varchar(50), site_id int, latitude_posted real, longitude_posted real);
CREATE TABLE subshift (subshift_id integer primary key autoincrement, employee_id int, subsite_id int, timestamp datetime, latitude real, longitude real);
